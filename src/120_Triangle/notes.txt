    var recur func(i int, j int) int 
    recur = func(i int, j int) int {
        val := triangle[i][j]
        if i + 1 < len(triangle) {
            return val + min(recur(i+1, j), recur(i+1, j+1))
        }
        return val
    }
    if len(triangle) == 1 { return triangle[0][0]}
    return triangle[0][0] + min(recur(1, 0), recur(1, 1))
}
